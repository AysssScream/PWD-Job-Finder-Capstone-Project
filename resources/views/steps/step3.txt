                            <script>
                                function toggleEmploymentOptions() {
                                    var statusSelect = document.getElementById("employment-status");
                                    var employmentOptions = document.getElementById("employment-type-options");
                                    var employmentTypeSelect = document.getElementById("employment-type");

                                    if (statusSelect.value === "employed") {
                                        employmentOptions.style.display = "block";
                                        // Clear options
                                        employmentTypeSelect.innerHTML = "";
                                        // Add employed options
                                        var employedOptions = [{
                                                value: "wage",
                                                text: "Wage Employment"
                                            },
                                            {
                                                value: "self",
                                                text: "Self Employed"
                                            },
                                            {
                                                value: "other",
                                                text: "Others"
                                            },
                                        ];
                                        employedOptions.forEach(function(option) {
                                            var optionElement = document.createElement("option");
                                            optionElement.value = option.value;
                                            optionElement.textContent = option.text;
                                            employmentTypeSelect.appendChild(optionElement);
                                        });
                                    } else if (statusSelect.value === "unemployed") {
                                        employmentOptions.style.display = "block";
                                        // Clear options
                                        employmentTypeSelect.innerHTML = "";
                                        // Add unemployed options
                                        var unemployedOptions = [{
                                                value: "new-entrant",
                                                text: "New Entrant/Fresh Graduate"
                                            },
                                            {
                                                value: "finished-contract",
                                                text: "Finished Contract"
                                            },
                                            {
                                                value: "resigned",
                                                text: "Resigned"
                                            },
                                            {
                                                value: "retired",
                                                text: "Retired"
                                            },
                                            {
                                                value: "terminated-calamity",
                                                text: "Terminated/Laid off due to calamity",
                                            },
                                            {
                                                value: "terminated-local",
                                                text: "Terminated/Laid off (Local)",
                                            },
                                            {
                                                value: "terminated-abroad",
                                                text: "Terminated/Laid off (abroad)",
                                            },
                                            {
                                                value: "other",
                                                text: "Others"
                                            },
                                        ];
                                        unemployedOptions.forEach(function(option) {
                                            var optionElement = document.createElement("option");
                                            optionElement.value = option.value;
                                            optionElement.textContent = option.text;
                                            employmentTypeSelect.appendChild(optionElement);
                                        });
                                    } else {
                                        employmentOptions.style.display = "none";
                                    }
                                }
                            </script>
